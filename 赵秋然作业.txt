#define _CRT_SECURE_NO_DEPRECATE
#include <stdio.h>
#include<stdlib.h>
#define array_length 7//定义数组长度

//输入
void input(int array[])
{
	for (int i = 0; i < array_length; i++)
	{
		printf("请输入第%d个数:", i + 1);
		scanf("%d", &array[i]);
	}
}

//输出
void output(int array[])
{
	printf("数组为：");
	for (int i = 0; i < array_length; i++)
	{
		printf("%d \t", array[i]);
	}
	printf("\n");
}

//选择法升序排列
void SelectSort(int * ListData, int length)
{
	for (int i = 0; i <= length - 2; i++)
	{
		int k = i;
		for (int j = i + 1; j <= length - 1; j++)
		{
			if (ListData[k]>ListData[j])
			{
				k = j;
			}
		}
		if (k != i)
		{
			int tmp = ListData[i];
			ListData[i] = ListData[k];
			ListData[k] = tmp;
		}
	}
}

//折半法查找
int search(int array[], int length, int key){
	int low = 0;
	int high = length - 1;
	while (low <= high){
		int mid = (low + high) / 2;
		int mid_val = array[mid];
		if (mid_val<key)
			low = mid + 1;
		else if (mid_val>key)
			high = mid - 1;
		else
			return mid;
	}
	return -1;
}

//处理查询结果
void deal_search_result(int array[])
{
	printf("输入你想查找的数：");
	int x;//要查找的整数x
	scanf("%d", &x);
	int result = search(array, array_length, x);
	if (result!=-1)
		printf("%d是数组的第%d个元素\n", x, result);
	else
		printf("%d不在数组中\n", x);
}

//主函数
int main()
{
	int array[array_length]; //定义数组 
	input(array);
	output(array);
	while (1)
	{
		int choise;
		printf("\n\n");
		printf("现在给你三个选择：\n");
		printf("1.选择法排序\n");
		printf("2.折半查找\n");
		printf("3.输出数组全部元素\n");
		printf("现在你想干什么？\n");
		scanf("%d", &choise);
		printf("\n");
		switch (choise)
		{
			case 1:
				SelectSort(array, array_length);
				printf("排序完成！\n");
				output(array);
				break;
			case 2:
				deal_search_result(array);
				break;
			case 3:
				output(array);
				break;
		default:
			break;
		}
		int another_choise;
		printf("\n想继续扣1：");
		scanf("%d", &another_choise);
		if (another_choise != 1)
			break;
		system("cls");
	}
	return 0;
}
